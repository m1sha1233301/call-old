function initSocket() { socket = io({ transports: ['websocket', 'polling'] }); socket.on('connect', () => { console.log('Connected to server'); socket.emit('register', currentUser); }); socket.on('user-list', (users) => { updateUsersList(users); }); socket.on('incoming-call', (data) => { handleIncomingCall(data); }); socket.on('call-accepted', async (data) => { console.log('Call accepted by:', data.targetName); document.getElementById('callStatus').textContent = `Звонок с ${data.targetName}`; }); socket.on('call-rejected', (data) => { alert(data.message); cleanupCall(); }); socket.on('call-cancelled', () => { closeIncomingCallModal(); alert('Звонок отменён'); }); socket.on('call-timeout', () => { alert('Время ожидания истекло'); cleanupCall(); }); socket.on('offer', async (data) => { await handleOffer(data.offer, data.from); }); socket.on('answer', async (data) => { await handleAnswer(data.answer); }); socket.on('ice-candidate', async (data) => { if (peerConnection && data.candidate) { try { await peerConnection.addIceCandidate(new RTCIceCandidate(data.candidate)); } catch (error) { console.error('Error adding ICE candidate:', error); } } }); socket.on('call-ended', () => { handleCallEnded(); }); socket.on('call-error', (data) => { alert(data.message); cleanupCall(); }); socket.on('disconnect', () => { console.log('Disconnected from server'); }); } function updateUsersList(users) { const usersList = document.getElementById('usersList'); const searchTerm = document.getElementById('searchInput').value.toLowerCase(); const filteredUsers = users.filter(user => user.userId !== currentUser.userId && user.username.toLowerCase().includes(searchTerm) ); usersList.innerHTML = filteredUsers.map(user => `
${user.username.charAt(0).toUpperCase()}
${user.username}
${user.inCall ? 'В звонке' : 'В сети'}
${!user.inCall ? `
` : ''}
`).join(''); } document.getElementById('searchInput').addEventListener('input', () => { socket.emit('get-users'); }); async function startCall(targetId, targetName, callType) { if (currentCall) { alert('Вы уже находитесь в звонке'); return; } currentCall = { targetId: targetId, targetName: targetName, callType: callType, isInitiator: true }; try { await initLocalStream(callType); await createPeerConnection(targetId); socket.emit('call-user', { targetId: targetId, callType: callType }); showVideoContainer(); document.getElementById('callStatus').textContent = `Звонок ${targetName}...`; } catch (error) { console.error('Error starting call:', error); alert('Ошибка запуска звонка: ' + error.message); cleanupCall(); } } async function initLocalStream(callType) { try { const constraints = { audio: true, video: callType === 'video' ? { width: { ideal: 1280 }, height: { ideal: 720 } } : false }; localStream = await navigator.mediaDevices.getUserMedia(constraints); document.getElementById('localVideo').srcObject = localStream; if (callType === 'audio') { document.getElementById('localVideo').style.display = 'none'; } } catch (error) { console.error('Error accessing media devices:', error); throw new Error('Не удалось получить доступ к камере/микрофону'); } } async function createPeerConnection(targetId) { peerConnection = new RTCPeerConnection(configuration); if (localStream) { localStream.getTracks().forEach(track => { peerConnection.addTrack(track, localStream); }); } peerConnection.ontrack = (event) => { console.log('Received remote track:', event.track.kind); const remoteVideo = document.getElementById('remoteVideo'); if (!remoteVideo.srcObject) { remoteVideo.srcObject = new MediaStream(); } remoteVideo.srcObject.addTrack(event.track); }; peerConnection.onicecandidate = (event) => { if (event.candidate) { socket.emit('ice-candidate', { targetId: targetId, candidate: event.candidate }); } }; peerConnection.oniceconnectionstatechange = () => { console.log('ICE connection state:', peerConnection.iceConnectionState); }; peerConnection.onconnectionstatechange = () => { console.log('Connection state:', peerConnection.connectionState); if (peerConnection.connectionState === 'connected') { document.getElementById('callStatus').textContent = 'Соединение установлено'; setTimeout(() => { document.getElementById('callStatus').style.display = 'none'; }, 2000); } else if (peerConnection.connectionState === 'disconnected' || peerConnection.connectionState === 'failed' || peerConnection.connectionState === 'closed') { handleCallEnded(); } }; peerConnection.onnegotiationneeded = async () => { try { if (currentCall && currentCall.isInitiator) { console.log('Creating offer...'); const offer = await peerConnection.createOffer({ offerToReceiveAudio: true, offerToReceiveVideo: true }); await peerConnection.setLocalDescription(offer); socket.emit('offer', { targetId: targetId, offer: offer }); } } catch (error) { console.error('Error during negotiation:', error); } }; } function handleIncomingCall(data) { if (currentCall) { socket.emit('reject-call', { fromId: data.from }); return; } incomingCallData = data; document.getElementById('callerAvatar').textContent = data.fromName.charAt(0).toUpperCase(); document.getElementById('callerName').textContent = data.fromName; document.getElementById('callType').textContent = data.callType === 'video' ? '📹 Видеозвонок' : '📞 Аудиозвонок'; document.getElementById('incomingCallModal').classList.add('active'); } async function acceptCall() { if (!incomingCallData) return; currentCall = { targetId: incomingCallData.from, targetName: incomingCallData.fromName, callType: incomingCallData.callType, isInitiator: false }; closeIncomingCallModal(); try { await initLocalStream(incomingCallData.callType); await createPeerConnection(incomingCallData.from); socket.emit('accept-call', { fromId: incomingCallData.from }); showVideoContainer(); document.getElementById('callStatus').textContent = `Звонок с ${incomingCallData.fromName}`; } catch (error) { console.error('Error accepting call:', error); alert('Ошибка при приёме звонка: ' + error.message); cleanupCall(); } incomingCallData = null; } function rejectCall() { if (incomingCallData) { socket.emit('reject-call', { fromId: incomingCallData.from }); incomingCallData = null; } closeIncomingCallModal(); } function closeIncomingCallModal() { document.getElementById('incomingCallModal').classList.remove('active'); } async function handleOffer(offer, from) { console.log('Received offer from:', from); if (!peerConnection) { console.error('PeerConnection not initialized'); return; } try { await peerConnection.setRemoteDescription(new RTCSessionDescription(offer)); console.log('Remote description set'); const answer = await peerConnection.createAnswer(); await peerConnection.setLocalDescription(answer); console.log('Local description set, sending answer'); socket.emit('answer', { targetId: from, answer: answer }); } catch (error) { console.error('Error handling offer:', error); } } async function handleAnswer(answer) { console.log('Received answer'); try { await peerConnection.setRemoteDescription(new RTCSessionDescription(answer)); console.log('Remote description set from answer'); } catch (error) { console.error('Error handling answer:', error); } } function toggleMute() { if (!localStream) return; const audioTrack = localStream.getAudioTracks()[0]; if (audioTrack) { audioTrack.enabled = !audioTrack.enabled; isMuted = !audioTrack.enabled; const muteBtn = document.getElementById('muteBtn'); muteBtn.classList.toggle('active', isMuted); muteBtn.textContent = isMuted ? '🔇' : '🎤'; } } function toggleVideo() { if (!localStream) return; const videoTrack = localStream.getVideoTracks()[0]; if (videoTrack) { videoTrack.enabled = !videoTrack.enabled; isVideoOff = !videoTrack.enabled; const videoBtn = document.getElementById('videoBtn'); videoBtn.classList.toggle('active', isVideoOff); videoBtn.textContent = isVideoOff ? '📷' : '📹'; } } function endCall() { if (currentCall) { socket.emit('end-call', { targetId: currentCall.targetId }); } cleanupCall(); } function handleCallEnded() { alert('Звонок завершён'); cleanupCall(); } function cleanupCall() { if (localStream) { localStream.getTracks().forEach(track => track.stop()); localStream = null; } if (peerConnection) { peerConnection.close(); peerConnection = null; } remoteStream = null; currentCall = null; isMuted = false; isVideoOff = false; document.getElementById('videoContainer').style.display = 'none'; document.getElementById('callScreen').querySelector('.empty-state').style.display = 'block'; document.getElementById('localVideo').srcObject = null; document.getElementById('remoteVideo').srcObject = null; document.getElementById('localVideo').style.display = 'block'; document.getElementById('callStatus').style.display = 'block'; const muteBtn = document.getElementById('muteBtn'); const videoBtn = document.getElementById('videoBtn'); muteBtn.classList.remove('active'); videoBtn.classList.remove('active'); muteBtn.textContent = '🎤'; videoBtn.textContent = '📹'; } function showVideoContainer() { document.getElementById('callScreen').querySelector('.empty-state').style.display = 'none'; document.getElementById('videoContainer').style.display = 'block'; } window.addEventListener('beforeunload', () => { if (currentCall) { socket.emit('end-call', { targetId: currentCall.targetId }); } });
